int search(std::vector<int>& nums, int target) {
    int start = 0, end = nums.size();
    while (start < end) {
        int middle = start + (end - start) / 2;
        if (nums[middle] > target) {
            end = middle;
        } else if (nums[middle] == target) {
            return middle;
        } else {
            start = middle + 1;
        }
    }
    return -1;
}
